for "(\"2019-nCoV\" OR \"COVID-19\" OR \"SARS-CoV-2\" OR \"HCoV-2019\" OR \"hcov\" OR \"NCOVID-19\" OR \"severe acute respiratory syndrome coronavirus 2\" OR \"severe acute respiratory syndrome corona virus 2\" OR ((\"coronavirus\" OR \"corona virus\") AND (Wuhan OR China OR novel))) AND (\"observational stud*\" OR cohort* OR \"non random*\" OR nonrandom*)"

return clinical_trials[abstract+acronym+conditions+date+date_inserted+gender+id+interventions+linkout+phase+registry+title]

sort by date_inserted desc
